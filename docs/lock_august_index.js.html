/**
 * Handles authentication logic.
 * @module Controller/lock/august/index
 * @exports Controller/lock/august/index
 */
const https = require('https');

/**
 * Retrieves the lock status of a device from the August API.
 * 
 * @param {string} deviceId - The ID of the device.
 * @param {Object} auth - The authentication object containing the token.
 * @returns {Promise<Object>} A promise resolving to the lock status response.
 */
const lockStatus = (deviceId, auth) => new Promise((resolve, reject) => {
    let opts = {
        hostname: 'api-production.august.com',
        port: 443,
        path: `/remoteoperate/${deviceId}/status`,
        method: 'PUT',
        headers: {
            'Content-Type': 'application/json',
            'x-august-api-key': '79fd0eb6-381d-4adf-95a0-47721289d1d9',
            'x-kease-api-key': '79fd0eb6-381d-4adf-95a0-47721289d1d9',
            'x-august-access-token': auth.token,
            'Accept-Version': '0.0.1',
            'User-Agent': 'August/Luna-3.2.2'
        }
    }
    let httpreq = https.request(opts, response => {
        let result = '';
        response.on('data', chunk => {
            result += chunk;
        });
        response.on('end', () => {
            try {
                const json = JSON.parse(result)
                resolve(json) 
            } catch (err) {
                reject(err)
            }
        });
    })

    httpreq.on('error', error => {
        reject(error);
    })

    httpreq.end()
})

/**
 * Sends a command to a device using the August API.
 * 
 * @param {string} deviceId - The ID of the device.
 * @param {string} command - The command to send (e.g., 'lock', 'unlock').
 * @param {Object} auth - The authentication object containing the token.
 * @returns {Promise<Object>} A promise resolving to the command response.
 */
const sendCommand = (deviceId, command, auth) => new Promise((resolve, reject) => {
    let opts = {
        hostname: 'api-production.august.com',
        port: 443,
        path: `/remoteoperate/${deviceId}/${command}`,
        method: 'PUT',
        headers: {
            'Content-Type': 'application/json',
            'x-august-api-key': '79fd0eb6-381d-4adf-95a0-47721289d1d9',
            'x-kease-api-key': '79fd0eb6-381d-4adf-95a0-47721289d1d9',
            'x-august-access-token': auth.token,
            'Accept-Version': '0.0.1',
            'User-Agent': 'August/Luna-3.2.2'
        }
    }
    let httpreq = https.request(opts, response => {
        let result = '';
        response.on('data', d => {
            result += d;
        })
        response.on('end', () => {
            try {

            }

});

/**
 * Retrieves logs for a device from the August API.
 * 
 * @param {string} deviceId - The ID of the device.
 * @param {Object} auth - The authentication object containing the token.
 * @returns {Promise<Object>} A promise resolving to the logs response.
 */
const getLogs = (deviceId, auth) => new Promise((resolve, reject) => {
    let datetime = new Date(new Date().getTime() + 86400000).toISOString();
    let opts = {
        hostname: 'api-production.august.com',
        port: 443,
        path: `/locks/log/after/${deviceId}/${datetime}/100`,

        method: 'GET',
        headers: {
            'Content-Type': 'application/json',
            'x-august-api-key': '79fd0eb6-381d-4adf-95a0-47721289d1d9',
            'x-kease-api-key': '79fd0eb6-381d-4adf-95a0-47721289d1d9',
            'x-august-access-token': auth.token,
            'Accept-Version': '0.0.1',
            'User-Agent': 'August/Luna-3.2.2'
        }
    }

    let httpreq = https.request(opts, response => {
        let result = '';
        response.on('data', chunk => {
            result += chunk.toString();
        })

        response.on('end', () => {
            let json = JSON.parse(result);
            resolve(json);
        })
    })

    httpreq.on('error', error => {
        reject([]);
    })

    httpreq.end();
});
