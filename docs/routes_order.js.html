/**
 * This module sets up routes for managing orders, processing payments, and handling subscriptions.
 * It includes routes for creating, retrieving, updating, and deleting orders, as well as managing payment sessions and subscriptions.
 * Stripe API is used for handling payments and subscriptions.
 *
 * @module routes/order
 * @exports router
 */
const router = require('express').Router();
const stripe = require('stripe')(process.env.stripe_sk_live)
const { authMiddleware } = require('../middlewares/authMiddleware');
const Order = require('../models/order')
const User = require('../models/user')
const Kit = require('../models/kit')
const PaymentMethod = require('../models/paymentMethod')
const OrderController = require('../controllers/order');
const { cognitoUserGroups } = require('../constants');

/**
 * Retrieves all orders associated with the authenticated user.
 * Requires authentication and returns orders in a JSON format.
 * 
 * @name GetUserOrders
 * @function
 * @memberof module:routes/order
 * @param {Object} req - The HTTP request object, containing user identification.
 * @param {Object} res - The HTTP response object used to return data or errors.
 */
router.get('/', authMiddleware, async (req, res) => {
  const { _id: userId } = req.user
  try {
    const userOrders = await Order.find({ user: userId }).lean()
    res.status(200).json(userOrders)
  } catch (error) {
    res.status(401).json({ message: "Order not found", error })
  }
})

/**
 * Retrieves all orders if the authenticated user is an admin.
 * Requires admin role and returns all orders in a JSON format.
 * 
 * @name GetAllOrders
 * @function
 * @memberof module:routes/order
 * @param {Object} req - The HTTP request object, indicating the user's role.
 * @param {Object} res - The HTTP response object for output.
 */
router.get('/all', authMiddleware, async (req, res) => {
  const { role } = req.user
  if (role !== 'admin') res.status(400).send('Only admins can have access to this route')
  try {
    const orders = await Order.find().lean()
    res.status(200).json(orders)
  } catch (error) {
    res.status(401).json({ message: "Order not found", error })
  }
})

/**
 * Retrieves all items within an order specified by the order ID in the URL.
 * Requires authentication and returns order items in a JSON format.
 * 
 * @name GetOrderItems
 * @function
 * @memberof module:routes/order
 * @param {Object} req - The HTTP request object, containing order ID in URL parameters.
 * @param {Object} res - The HTTP response object for output.
 */
router.get('/:order/items', authMiddleware, async (req, res) => {
  const { order } = req.params
  try {
    const items = await Kit.find({ order }).lean()
    res.status(200).json(items)
  } catch (error) {
    res.status(401).json({ message: "Order not found", error })
  }
})

/**
 * Lists all orders with detailed user information, accessible only by admins or superadmins.
 * 
 * @name ListOrdersWithUserInfo
 * @function
 * @memberof module:routes/order
 * @param {Object} req - The HTTP request object, specifying admin-level access.
 * @param {Object} res - The HTTP response object to send back user information.
 */
router.get('/admin/list', authMiddleware, async (req, res) => {
  const { role } = req.user
  if (![cognitoUserGroups.admin,cognitoUserGroups.superadmin].includes(role)) res.status(400).send('Only admins can have access to this route')
  try {
    const orders = await Order.find()
      .populate('user', 'firstName lastName company email ')
      .lean();
    res.status(200).json(orders)
  } catch (error) {
    res.status(401).json({ message: "Order not found", error })
  }
})

/**
 * Retrieves a single order by its ID.
 * Requires authentication and returns the order details in a JSON format.
 * 
 * @name GetOrderById
 * @function
 * @memberof module:routes/order
 * @param {Object} req - The HTTP request object, containing the order ID as a path parameter.
 * @param {Object} res - The HTTP response object for returning the order details.
 */
router.get('/:orderId', authMiddleware, async (req, res) => {
  const { orderId } = req.params
  try {
    const order = await Order.findById(orderId).lean()
    res.status(200).json(order)
  } catch (error) {
    res.status(401).json({ message: "Order not found", error })
  }
})

/**
 * Creates a new order and initiates a Stripe payment session if necessary.
 * Handles both one-time payments and setup for future recurring payments.
 * 
 * @name CreateOrder
 * @function
 * @memberof module:routes/order
 * @param {Object} req - The HTTP request object, containing the user and order details.
 * @param {Object} res - The HTTP response object for sending back the response with Stripe session details or errors.
 */
router.post('/', authMiddleware, async (req, res) => {
  const { _id: userId, email: clientEmail } = req.user
  const order = req.body
  const { paymentMethod, details } = order
  const responseObject = {
    message: 'Order created successfully'
  }
  try {
    const user = await User.findById(userId).select('firstName lastName email phone role stripeCustomerId')

    if (!user.stripeCustomerId) {
      const customer = await stripe.customers.create({
        email: user.email,
      });
      user.stripeCustomerId = customer.id;
      await user.save();
    }

    const { stripeCustomerId: customer } = user


    await OrderController.createAndNotify({
      ...order,
      user: userId,
    }, user)

    if (['credit card', 'credit', 'card'].includes(paymentMethod)) {
      const userWantKits = details.some(request => request.serviceType === 'kit')
      const serviceRequests = details.filter(request => request.serviceType !== 'kit')

      let baseSessionConfig = {
        customer,
        client_reference_id: userId,
        billing_address_collection: 'required',
        customer_update: {
          address: 'auto',
        },
        consent_collection: {
          terms_of_service: 'required',
        },
        metadata: {
          userId,
          // orderId: newOrder._id.toString()
        },
        success_url: `${process.env.APP_URL ?? 'https://app.delet.ai'}/order/success`,
        cancel_url: `${process.env.APP_URL ?? 'https://app.delet.ai'}/kits/orders`,
      };

      if (serviceRequests.length > 0) {

        const checkoutConfig = {
          ...baseSessionConfig,
          payment_intent_data: {
            setup_future_usage: 'off_session',
          },
          line_items: serviceRequests.map(product => ({
            price: product.priceId,
            quantity: product.amount,
          })),
          mode: 'payment',
          custom_text: {
            terms_of_service_acceptance: {
              message: userWantKits
                ? "By accepting you agree to be charged for these products now and for the kits once they are delivered."
                : "By accepting you agree on Delet using this payment method for future operations",
            },
          },
        };

        const stripeSession = await stripe.checkout.sessions.create(checkoutConfig);
        responseObject.stripeUrl = stripeSession.url

      } else if (userWantKits) {
        const stripePaymentMethod = await PaymentMethod.findOne({
          user: userId,
        });

        if (!stripePaymentMethod) {

          const checkoutConfig = {
            ...baseSessionConfig,
            mode: 'setup',
            payment_method_types: ['card'],
            custom_text: {
              terms_of_service_acceptance: {
                message: "By accepting you understand that you will be charged once you get the first kit or set of kits",
              },
            },
            setup_intent_data: {
              metadata: {
                userId,
                // orderId: newOrder._id.toString(),
                paymentMethod,
                status: 'pending'
              }
            },

          };

          const stripeSession = await stripe.checkout.sessions.create(checkoutConfig);

          responseObject.stripeUrl = stripeSession.url
        }
      }
    }

    res.status(200).json(responseObject);

  } catch (error) {
    console.error('Failed to create payment link:', error);
    res.status(500).json({ error: error.message });
  }
});

/**
 * Retrieves session information from Stripe based on a session ID provided as a query parameter.
 * Requires authentication.
 * 
 * @name GetSessionInfo
 * @function
 * @memberof module:routes/order
 * @param {Object} req - The HTTP request object, containing the session ID as a query parameter.
 * @param {Object} res - The HTTP response object for sending back session details.
 */
router.get('/session-info', authMiddleware, async (req, res) => {
  const sessionId = req.query.payment_id;

  try {
    const session = await stripe.checkout.sessions.retrieve(sessionId);
    res.json({ session: session });
  } catch (error) {
    console.error('Failed to create payment link:', error);
    res.status(500).json({ error: error.message });
  }
});


/**
 * Retrieves subscription information from Stripe based on a session ID provided as a query parameter.
 * Requires authentication.
 * 
 * @name GetSubscriptionInfo
 * @function
 * @memberof module:routes/order
 * @param {Object} req - The HTTP request object, containing the session ID as a query parameter.
 * @param {Object} res - The HTTP response object for sending back subscription details.
 */
router.get('/subscription-info', authMiddleware, async (req, res) => {
  const sessionId = req.query.payment_id;

  try {
    const session = await stripe.checkout.sessions.retrieve(sessionId);
    const subscription = await stripe.subscriptions.retrieve(session.subscription)
    res.json({ subscription: subscription });
  } catch (error) {
    console.error('Failed to create payment link:', error);
    res.status(500).json({ error: error.message });
  }
});


/**
 * Deletes all orders associated with the authenticated user.
 * Requires authentication and potentially admin privileges depending on business rules.
 * 
 * @name DeleteAllOrders
 * @function
 * @memberof module:routes/order
 * @param {Object} req - The HTTP request object, containing the user's identification.
 * @param {Object} res - The HTTP response object for sending back the deletion results.
 */
router.delete('/all', authMiddleware, async (req, res) => {
  // if (req.user.role !== 'admin') res.status(400).send('Only admins can have access to this route')
  const { _id: userId } = req.user

  try {
    const deletedResponse = await Order.deleteMany({ user: userId })
    await Kit.updateMany({}, { $set: { order: null } });
    res.status(200).json(deletedResponse)
  } catch (error) {
    res.status(401).json({ message: "Order not found", error })
  }
})

/**
 * Updates the state of an order specified by the order ID in the URL.
 * Accessible only by admins or superadmins.
 * 
 * @name UpdateOrderState
 * @function
 * @memberof module:routes/order
 * @param {Object} req - The HTTP request object, containing the current and new state of the order.
 * @param {Object} res - The HTTP response object for sending back the updated order details.
 */
router.patch('/:id/update-state', authMiddleware, async (req, res) => {
  if (![cognitoUserGroups.admin,cognitoUserGroups.superadmin].includes(req.user.role)) res.status(400).send('Only admins can have access to this route')

  const { currentState, newState } = req.body

  try {
    const order = await Order.findById(req.params.id).populate('user', 'email firstName lastName company')
    order.state = newState
    await order.save()
    res.status(200).json(order)
  } catch (error) {
    console.log(error)
  }
})
