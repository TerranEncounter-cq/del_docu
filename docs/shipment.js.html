/**
 * Handles shipment-related operations.
 * @module Controller/shipment
 */
const Shipment = require("../models/shipment");
const { sendEmail } = require("./notification");

/**
 * Sends a notification email for a shipment with formatted information.
 * 
 * @param {Object} shipment - The shipment object containing details about the shipment.
 */
const sendShipmentNotification = (shipment) => {
  const emailBody = `
  <html>
    <head>
      <style>
        body { font-family: Arial, sans-serif; }
        .email-container { max-width: 600px; margin: auto; border: 1px solid #ccc; padding: 20px; }
        .header { background-color: #f2f2f2; padding: 10px; text-align: center; }
        .content { margin-top: 20px; }
        .footer { margin-top: 20px; font-size: 12px; text-align: center; color: #777; }
        .info-section { margin-bottom: 20px; }
        .info-label { font-weight: bold; }
      </style>
    </head>        
    ${formattedShipment(shipment)} <br>
  </html>`;
  try {
    sendEmail({
      to: 'logistics@delet.com',
      subject: `A shipment has been ${shipment.state}`,
      body: emailBody

    })
  } catch (error) {
    console.log(`There's been an error: ${error}`)
    throw new Error(error)
  }
}

/**
 * Formats the shipment details into an HTML structure for email.
 * 
 * @param {Object} shipment - The shipment object containing details about the shipment.
 * @returns {string} - The formatted HTML content representing the shipment details.
 */
const formattedShipment = (shipment) => {
  return `
    <body>
      <div class="email-container">
        <div class="header">
          <h2>A shipment has been ${shipment.state}</h2>
        </div>
        <div class="content">
          <div class="info-section">
            <div class="info-label">Shipment Code:</div>
            <div>${shipment.code}</div>
          </div>

          ${shipment.state === 'completed' && `
            <div class="info-section">
              <div class="info-label">Click this link to confirm you received the kits so you can use them</div>
              <a href="${process.env.APP_URL}/shipment/${shipment._id}/confirm">Confirm shipment</a>
              <span>You can also do from your <a href="${process.env.APP_URL}/kits/orders">orders </a>list</span>
            </div>
          `}

          <div class="info-section">
            <div class="info-label">Status:</div>
            <div>${shipment.state}</div>
          </div>

          ${shipment.estimatedDelivery ? `<div class="info-section">
          <div class="info-label">Estimated Delivery:</div>
          <div>MM/DD/YYYY</div>
        </div>` : ''}
          <div class="info-section">
            <div class="info-label">Delivery Address:</div>
            <div>${shipment.deliveryAddress.street} ${shipment.deliveryAddress.city} ${shipment.deliveryAddress.state} ${shipment.deliveryAddress.zipCode}</div>
          </div>
          <div class="info-section">
            <div class="info-label">Tracking URL:</div>
            <a href="${shipment.trackingUrl}">Track your shipment</a>
          </div>
          ${shipment.notes ? `<div class='info-section'>
          <div class='info-label'>Notes:</div>
          <div>${shipment.notes}</div>
        </div> `: ''}
          
        </div>
        <div class="footer">
          <p>Thank you for choosing Our Company.</p>
        </div>
      </div>
    </body>
  
  `;
};

module.exports = {
  sendShipmentNotification
}
