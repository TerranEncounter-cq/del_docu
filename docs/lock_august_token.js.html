/**
 * Handles authentication logic.
 * @module Controller/lock/august/token
 * @exports Controller/lock/august/token
 */
const https = require('https');
const { v4: uuidv4 } = require('uuid');

/**
 * Initiates a session with the August API to authenticate a user.
 * 
 * @param {Object} req - The request object containing user credentials.
 * @param {Object} res - The response object to send back the session token.
 */
function session(req, res) {
    const { email, password } =  req.body;
    const data = JSON.stringify({
        "installId": uuidv4(),
        "password": password,
        "identifier": `email:${email}`
    });

    const opts = {
        hostname: 'api-production.august.com',
        port: 443,
        path: '/session',
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'x-august-api-key': '79fd0eb6-381d-4adf-95a0-47721289d1d9',
            'x-kease-api-key': '79fd0eb6-381d-4adf-95a0-47721289d1d9',
            'Accept-Version': '0.0.1',
            'User-Agent': 'August/Luna-3.2.2',
            'Content-Length': data.length
        }
    }

    let responseBody = '';

    const request = https.request(opts, response => {
        const token = response.headers['x-august-access-token'];
        response.on('data', chunk => {
            responseBody += chunk;
        });
        response.on('end', function () {
            const json = JSON.parse(responseBody)
            res.status(201).json({ 
                token: token,
                userId: json.userId,
                iat: json.iat,
                exp: json.exp
            });
        });
    });

    request.on('error', error => {
        res.status(500).send(error);
    });


    request.write(data);

    request.end();
    
}

/**
 * Validates user information using the August API.
 * 
 * @param {Object} req - The request object containing user information and token.
 * @param {Object} res - The response object to send back the validation result.
 */
function validate(req, res) {
    const { username, email, token } =  req.body;

    const data = JSON.stringify({
        "username": username,
        "value" : email        
    });
    
    const opts = {
        hostname: 'api-production.august.com',
        port: 443,
        path: '/validation/email',
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'x-august-api-key': '79fd0eb6-381d-4adf-95a0-47721289d1d9',
            'x-kease-api-key': '79fd0eb6-381d-4adf-95a0-47721289d1d9',
            'x-august-access-token': token,
            'Accept-Version': '0.0.1',
            'User-Agent': 'August/Luna-3.2.2',
            'Content-Length': data.length
        }
    }

    let responseBody = '';

    const request = https.request(opts, response => {
        response.on('data', chunk => {
            responseBody += chunk;
        });
        response.on('end', function () {
            res.status(201).json(JSON.parse(responseBody));
        });
    });

    request.on('error', error => {
        res.status(500).send(error);
    });

    request.write(data);

    request.end();
}
