/**
 * Handles authentication logic.
 * @module Controller/lock/index
 * @exports Controller/lock/index
 */
const ObjectId = require('mongoose').Types.ObjectId;
const { v4: uuidv4 } = require('uuid');
const augustController = require('./august/index');
const switchbotController = require('../../helpers/switchbot');
const { createTransaction } = require('../transactions');
const Hardware = require('../../models/hardware');
const Meeting = require('../../models/meeting');

// constants for 'Ensure_lock'
const https = require('https');
const crypto = require('crypto');
const uuid = require('uuid');

/**
 * Toggles the lock status of a device.
 * 
 * @param {string} deviceId - The ID of the device.
 * @param {string} command - The command to perform (e.g., 'lock', 'unlock').
 * @param {Object} auth - The authentication object containing token, secret, t, and nonce.
 * @param {string} [parameter='default'] - The parameter for the command (optional).
 * @returns {Promise<Object>} A promise resolving to the API response.
 */
async function toggleLock(deviceId, command, auth, parameter = 'default')
{
    const sign = generateSign(auth);

    const data = JSON.stringify({
        "command": command,
        "parameter": parameter,
        "commandType": "command"
    });

    const options = {
        method: 'POST',
        headers: {
            "Authorization": auth.token,
            "sign": sign,
            "nonce": auth.nonce,
            "t": auth.t,
            'Content-Type': 'application/json',
            "charset": 'utf8',
        },
        body: data
    };

    const hostname = 'https://api.switch-bot.com';
    const path = `v1.1/devices/${deviceId}/commands`;
    const port = 443;
    const url = `${hostname}:${port}/${path}`;

    const apiResponse = await fetch(url, options); // cannot rely on this for valid response.
    const apiResult = await apiResponse.json();
    return apiResult;
}

/**
 * Generates a sign for authentication.
 * 
 * @param {Object} param0 - The object containing token, secret, t, and nonce.
 * @returns {string} The generated sign.
 */
const generateSign = ({ token, secret, t, nonce }) => {
    const data = token + t + nonce;
    const signTerm = crypto.createHmac('sha256', secret).update(Buffer.from(data, 'utf-8')).digest();
    return signTerm.toString("base64");
}

/**
 * Retrieves the status of a device.
 * 
 * @param {string} deviceId - The ID of the device.
 * @param {Object} auth - The authentication object containing token, secret, t, and nonce.
 * @returns {Promise<Object>} A promise resolving to the device status.
 */
const getDevicesStatus = (deviceId, auth) => new Promise((resolve, reject) => {
    const sign = generateSign(auth);

    const options = {
        hostname: 'api.switch-bot.com',
        port: 443,
        path: `/v1.1/devices/${deviceId}/status`,
        method: 'GET',
        headers: {
            "Authorization": auth.token,
            "sign": sign,
            "nonce": auth.nonce,
            "t": auth.t,
            'Content-Type': 'application/json',
        },
    };

    const req = https.request(options, res => {
        res.setEncoding('utf8');

        let body = '';

        res.on('data', function (chunk) {

            body = body + chunk;
        });

        res.on('end', function () {
            try {
                let json = JSON.parse(body);
                resolve(json);
            } catch (e) {
                reject(e);
            }
        });
    });

    req.on('error', error => {
        reject(error);
    });

    req.end();
});

/**
 * Retrieves the lock status of a hardware item.
 * 
 * @param {Object} req - The request object containing the hardware ID in params.
 * @param {Object} res - The response object.
 * @param {Function} next - The next middleware function.
 */
function getStatus(req, res, next) {
    if (!ObjectId.isValid(req.params.id)) {
        return res.status(400).end();
    }

    let query = {
        _id: req.params.id,
        category: 'lock',
        status: 'active'
    }
    
    Hardware.findOne(query).populate('account').exec(function (err, hardware) {

        if (!hardware || !hardware.account) {
            return res.status(403).end();
        }
        if (hardware.lock.vendor === 'august') {
            const authData = {
                token: hardware.account.token,
            }
            augustController.lockStatus(hardware.lock.id, authData)
                .then((response) => {
                    let lockStatus = 'Unknown';
                    if (response.status.toLowerCase() == 'kauglockstate_locked') {
                        lockStatus = 'Locked';
                    }

                    if (response.status.toLowerCase() == 'kauglockstate_unlocked') {
                        lockStatus = 'Open';
                    }

                    res.json(lockStatus);

                })
                .catch((err) => {
                    console.log(err);
                    res.json('Unknown');
                });
        } else if (hardware.lock.vendor === 'switchbot') {
            let lockStatus = 'Unknown';
            const authData = {
                token: hardware.account.token,
                secret: hardware.account.secret,
                t: Date.now(),
                nonce: uuidv4()
            }
            switchbotController.getDevicesStatus(hardware.lock.id, authData)
                .then((response) => {
                    if (response.body.lockState === 'locked') {
                        lockStatus = 'Locked';
                    }

                    if (response.body.lockState === 'unlocked') {
                        lockStatus = 'Open';
                    }

                    res.json(lockStatus);
                })
                .catch((err) => {
                    console.log(err);
                    res.json('Unknown');
                });
        }

    });
}

/**
 * Retrieves logs related to the lock of a hardware item.
 * 
 * @param {Object} req - The request object containing the hardware ID in params.
 * @param {Object} res - The response object.
 * @param {Function} next - The next middleware function.
 */
function getLogs(req, res, next) {

    if (!ObjectId.isValid(req.params.id)) {
        return res.status(400).end();
    }

    let query = {
        _id: req.params.id,
        user: req.user._id,
        category: 'lock',
        status: 'active'
    }
    if (req.user.orgs) {
        query.user = req.user.orgs;
    }
    Hardware.findOne(query).populate('account').exec(function (err, hardware) {

        if (!hardware || !hardware.account) {
            return res.status(403).end();
        }

        const authData = {
            token: hardware.account.token,
        }
        augustController.getLogs(hardware.lock.id, authData)
            .then((response) => {
                res.json(response)
            })
            .catch((err) => {
                res.json(err)
            })
    });


}

/**
 * Sends a command to a hardware device to perform a lock/unlock action.
 * 
 * @param {Object} req - The request object containing the hardware ID and action in params.
 * @param {Object} res - The response object.
 * @param {Function} next - The next middleware function.
 */
function sendCommand(req, res, next) {
    if (!ObjectId.isValid(req.params.id)) {
        return res.status(400).end();
    }

    let query = {
        _id: req.params.id,
        user: req.user._id,
        category: 'lock',
        status: 'active'
    }

    if (req.user.orgs) {
        query.user = req.user.orgs;
    }

    Hardware.findById(req.params.id).populate('account').exec(async function (err, hardware) {
        
        if (!hardware) {
            return res.status(403).end();
        }
        
        if (hardware.lock.vendor === 'august') {
            const authData = {
                token: hardware.account.token,
            }
            const command = req.params.action.toLowerCase();
            augustController.sendCommand(hardware.lock.id, command, authData)
                .then(response => {
                    if ((response.status === 'kAugLockState_Unlocked' && command === 'unlock') || (response.status === 'kAugLockState_Locked' && command === 'lock')) {
                        createTransaction({
                            action: command,
                            type: 'webpage',
                            byModel: 'User',
                            by: hardware.user,
                            hardware: hardware._id,
                            property: hardware.property
                        })
                    }
                    res.send(command);
                }).catch((err) => {
                    console.log(err);
                })
        } else if (hardware.lock.vendor === 'switchbot') {
            const deviceId = hardware.lock.id;
            const command = req.params.action.toLowerCase();
            const authData = {
                token: hardware.account.token,
                secret: hardware.account.secret,
                t: Date.now(),
                nonce: uuidv4()
            }

            /*
            * Ensure_lock
            * This function is used to ensure that the lock is in the desired state.
            * Status code 200: SUCCESS
            * Status code 490: TIMEOUT
            * Status code 491: HUB OFFLINE
            * Status code 492: UNKNOWN ERROR
            * Status code 493: FAILED TO LOCK/UNLOCK
            * Status code 494: UNKNOWN ERROR
            */
            try {
                const lockResponse = await toggleLock(deviceId, command, authData, 'default');

                if ('errorMessage' in lockResponse) {
                    if (lockResponse.errorMessage.includes('Task timed out')){
                        console.log("TIMEOUT: Request timed out");
                        return res.status(490).send("TIMEOUT: Request timed out");
                    }
                } else if ('statusCode' in lockResponse) {
                    if(lockResponse.statusCode == 1100) {
                        console.log("HUB OFFLINE");
                        return res.status(491).send("HUB OFFLINE");
                    }
                    else if(lockResponse.statusCode == 190) {
                        console.log("UNKNOWN ERROR");
                        return res.status(492).send("UNKNOWN ERROR");
                    }

                    else if(lockResponse.statusCode == 100) {
                        console.log("LOCK ONLINE: Communicated successfully with Switchbot API.");
                        if ('body' in lockResponse) {
                            if('items' in lockResponse.body) {
                                if(lockResponse.body.items.length > 0) {
                                    if ('status' in lockResponse.body.items[0]) {
                                        if ('lockState' in lockResponse.body.items[0].status) {
                                            if(lockResponse.body.items[0].status.lockState == "locked" && command == "lock") {
        
                                                console.log("LOCKED");
                                                let currentLockState = "";
                                                let i = 0;
        
                                                setTimeout(async function() {
                                                    console.log("Waiting for 4 seconds...");
        
                                                    while (currentLockState != "locked" && i < 100) {
                                                        statusResponse = await getDevicesStatus(deviceId, authData);
                                                        currentLockState = statusResponse.body.lockState;
                                                        console.log(currentLockState);
                                                        i += 1;
                                                    }
                                                    if(i == 100)
                                                        console.log("INTERNAL TIMEOUT: Hardware miscommunication");
                                                    
            
                                                    statusResponse = await getDevicesStatus(deviceId, authData);
                                                    currentLockState = statusResponse.body.lockState;
                                                    if (currentLockState == "unlocked") {
                                                        console.log(`ERR: Failed to lock. Current status is ${currentLockState}`);
                                                        return res.status(493).send(`ERR: Failed to lock. Current status is ${currentLockState}`);
                                                    } else if (currentLockState == "locked") {
                                                        console.log(`SUCCESS: Locked. Current status is ${currentLockState}`);
                                                        return res.status(200).send(`SUCCESS: Locked. Current status is ${currentLockState}`);
                                                    } else {
                                                        console.log(`ERR: Unknown error. Current status is ${currentLockState}`);
                                                        return res.status(494).send(`ERR: Unknown error. Current status is ${currentLockState}`);
                                                    }
                                                }, 4000);
                                            }
                                            else if (lockResponse.body.items[0].status.lockState == "unlocked" && command == "unlock") {
                                                console.log("UNLOCKED");
                                                let currentLockState = "";
                                                let i = 0;
        
                                                setTimeout(async function() {
                                                    console.log("Waiting for 4 seconds...");
        
                                                    while (currentLockState != "unlocked" && i < 100){
                                                        statusResponse = await getDevicesStatus(deviceId, authData);
                                                        currentLockState = statusResponse.body.lockState;
                                                        console.log(currentLockState);
                                                        i += 1;
                                                    }
                                                    if(i == 100)
                                                        console.log("Hardware miscommunication");
                                                    
            
                                                    statusResponse = await getDevicesStatus(deviceId, authData);
                                                    currentLockState = statusResponse.body.lockState;
                                                    if (currentLockState == "locked") {
                                                        console.log(`ERR: Failed to unlock. Current status is ${currentLockState}`);
                                                        return res.status(493).send(`ERR: Failed to unlock. Current status is ${currentLockState}`);
                                                    }
                                                    else if (currentLockState == "unlocked") {
                                                        console.log(`SUCCESS: Unlocked. Current status is ${currentLockState}`);
                                                        return res.status(200).send(`SUCCESS: Unlocked. Current status is ${currentLockState}`);
                                                    }
                                                    else {
                                                        console.log(`ERR: Unknown error. Current status is ${currentLockState}`);
                                                        return res.status(494).send(`ERR: Unknown error. Current status is ${currentLockState}`);
                                                    }
                                                }, 4000);
                                            }
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
                else{
                    const statusResponse = await getDevicesStatus(deviceId, authData);
                    const currentLockState = statusResponse.body.lockState;
                    console.log(currentLockState);
                    console.log(`UNKNOWN ERROR. Current Status ${currentLockState}`)
                }

            } catch (error) {
                console.error('Error in Ensure_lock:', error);
                throw error;
            }
        }
    });
}
