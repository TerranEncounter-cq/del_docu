/**
 * Handles the creation and updating of orders and notifies logistics via email.
 * @module Controller/order
 */
const Order = require("../models/order");
const { sendEmail } = require("./notification");

/**
 * Creates a new order and notifies logistics via email.
 * @param {Object} orderData - The data for the new order.
 * @param {Object} user - The user placing the order.
 * @returns {Promise} A promise that resolves with the created order or rejects with an error.
 */
const createAndNotify = async function (orderData, user) {
  const order = new Order(orderData);

  const emailBody = `
  <body>
      Payment Method: ${orderData.paymentMethod} <br><br>
      <h2>Order details</h2>
      ${formatOrderDetails(orderData.details)} <br>
      ${orderData.notes ? 'Notes: ' + orderData.notes : ''}
      <h2>User Information</h2>
      ${formatUserDetails(user)}
  </body>`;

  try {
    await order.save();

    sendEmail({
      to: 'logistics@delet.com',
      subject: `A new order with ID ${order._id} has been placed`,
      body: emailBody
    })

    return order
  } catch (error) {
    console.log(`There's been an error: ${error}`)
    throw new Error(error)
  }
};

 * Updates an existing order and notifies logistics via email.

/**
 * Formats the order details into an HTML table.
 * @param {Array} details - The order details.
 * @returns {string} The formatted HTML table representing the order details.
 */
const formatOrderDetails = (details) => {
  const rows = details.map(detail => `
      <tr>
          <td style="text-align: center; vertical-align: middle;">${detail.serviceType}</td>
          <td style="text-align: center; vertical-align: middle;">${detail.amount}</td>
          <td style="text-align: center; vertical-align: middle;">${detail.price}</td>
          <td style="text-align: center; vertical-align: middle;">${detail.notes || 'N/A'}</td>
      </tr>
  `).join('');
  return `
      <table border="1" style="width:100%; border-collapse: collapse;">
          <thead>
              <tr>
                  <th style="text-align: center; vertical-align: middle;">Service Type</th>
                  <th style="text-align: center; vertical-align: middle;">Amount</th>
                  <th style="text-align: center; vertical-align: middle;">Price</th>
                  <th style="text-align: center; vertical-align: middle;">Notes</th>
              </tr>
          </thead>
          <tbody>
              ${rows}
          </tbody>
      </table>
  `;
};

/**
 * Formats the user details into an HTML table.
 * @param {Object} user - The user details.
 * @returns {string} The formatted HTML table representing the user details.
 */
const formatUserDetails = (user) => {
  return `
    <table border="1" style="width:100%; border-collapse: collapse; margin-top: 20px;">
      <thead>
        <tr>
          <th style="text-align: center; vertical-align: middle;">First Name</th>
          <th style="text-align: center; vertical-align: middle;">Last Name</th>
          <th style="text-align: center; vertical-align: middle;">Email</th>
          <th style="text-align: center; vertical-align: middle;">Phone</th>
          <th style="text-align: center; vertical-align: middle;">Role</th>
        </tr>
      </thead>
      <tbody>
        <tr>
          <td style="text-align: center; vertical-align: middle;">${user.firstName}</td>
          <td style="text-align: center; vertical-align: middle;">${user.lastName}</td>
          <td style="text-align: center; vertical-align: middle;">${user.email}</td>
          <td style="text-align: center; vertical-align: middle;">${user.phone}</td>
          <td style="text-align: center; vertical-align: middle;">${user.role}</td>
        </tr>
      </tbody>
    </table>
  `;
};
