/**
 * Handles requests related to the dashboard, including retrieving statistics and recent leads.
 * @module Controller/dashboard
 */
const IDAnalyzer = require("idanalyzer");
const ObjectId = require("mongoose").Types.ObjectId;
const Contact = require("../models/contact");
const Property = require("../models/property");
const eventEmitter = require("../events/emitter");
const Meeting = require("../models/meeting");
const { ROWS_PER_PAGE } = require("../constants/index");

/**
 * Retrieves dashboard statistics including contacts count, tours count, 
 * and property details for the last 7 days, and sends them in the response.
 * @param {Object} req - The request object.
 * @param {Object} res - The response object.
 */
const getDashboardStats = async (req, res) => {
  let query = { $or: [
        { user: req.user._id },
        { allowedAgents: req.user._id  }
      ] };
  if (req.user.orgs) {
    query.user = req.user.orgs;
  }

  // Calculate the date 7 days ago from now
  const sevenDaysAgo = new Date();
  sevenDaysAgo.setDate(sevenDaysAgo.getDate() - 7);

  // Add the condition for the last 7 days
  query.createdDate = { $gte: sevenDaysAgo };

  const contactsCount = await Contact.find(query, {
    createdDate: 1,
    idImage: 1,
    firstName: 1,
    lastName: 1,
    phone: 1,
    email: 1,
    meetings: 1,
    idImage: 1,
  }).count();

  query.status = "attended";
  const toursCount = await Meeting.find(query, { createdDate: 1 }).count();

  const properties = await Property.aggregate([
    {
      $lookup: {
        from: "meetings", // assuming your meetings collection is named 'meetings'
        localField: "_id",
        foreignField: "property",
        as: "meetings",
      },
    },
    {
      $match: {
        "meetings.status": "attended", // adjust based on your 'completed' tour status
        status: { $ne: "rented" }, // exclude rented properties
        "meetings.startTime": { $gte: sevenDaysAgo.getTime() }, // Filter meetings within the last 7 days
      },
    },
    {
      $group: {
        _id: "$_id",
        completedTours: { $sum: 1 },
        property: { $first: "$$ROOT" },
      },
    },
    {
      $sort: { completedTours: -1 },
    },
    {
      $replaceRoot: { newRoot: "$property" },
    },
  ]);

  delete query.status;
  const constantQuery = await Contact.find({
    ...query,
    createdDate: { $gt: sevenDaysAgo },
  }).sort({ createdDate: -1 });

  res.status(200).json({
    properties,
    contactsCount,
    toursCount,
    recentLeads: constantQuery,
  });
};
module.exports = { getDashboardStats };
