/**
 * Handles the sending of email notifications for meetings, including reminders and instructions for upcoming events.
 * This module interacts with the Meeting model to update meeting records and uses configured notification services
 * to send emails. It ensures that meeting participants receive timely and relevant information about their scheduled meetings,
 * enhancing the user experience by providing clear, actionable communications.
 *
 * @module controllers/notifications
 * @exports sendReminder Sends a reminder email to a contact about an upcoming meeting.
 * @exports sendInstructions Sends detailed instructions via email to a contact for an upcoming meeting.
 */
const notification = require("../controllers/notification");
const { generateReminderEmail, generateInstructionsEmail } = require("./emailGeneration");

const Meeting = require('../models/meeting');

/**
 * Sends a reminder email to a contact about an upcoming meeting.
 * If the contact is subscribed, it constructs the email body using `generateReminderEmail` and sends it.
 * After sending the email, it updates the meeting document to mark the reminder as sent.
 *
 * @param {Object} meeting - The meeting object containing contact, property, and user details.
 */
async function sendReminder(meeting) {
  const emailBody = generateReminderEmail(meeting.contact, meeting, meeting.property);
  const companyName = meeting.user.company.name || 'Delet';

  try {
    if (meeting.contact.subscription) {
      await notification.sendEmail({
          emailAlias: companyName,
          to: meeting.contact.email,
          subject: `[Reminder] ${companyName} House tour`,
          body: emailBody
      });

      await Meeting.updateOne({ _id: meeting._id }, {
          $set: { 'reminder.sent': true}
      });
      console.log('Meeting reminder sent for:', meeting._id);
  }
  } catch (error) {
    
  }

  
}

/**
 * Sends detailed instructions via email to a contact for an upcoming meeting.
 * It generates a random lock key, constructs the email using `generateInstructionsEmail`, and sends it if the contact is subscribed.
 * Additionally, it updates the meeting document to store the lock key and marks the instructions as sent.
 *
 * @param {Object} meeting - The meeting object containing contact, property, user details, and subscription information.
 */
async function sendInstructions(meeting) {
  const lockKey = Math.round(Math.random() * 10000000);
  const link = `https://app.delet.ai/meeting/unlock/${meeting._id}/${lockKey}`;
  const companyName = meeting.user.company.name || 'Delet';

  await Meeting.updateOne({ _id: meeting._id }, { $set: { lockKey } });

  const emailBody = generateInstructionsEmail(meeting.contact, meeting, meeting.property)

  if (meeting.contact.subscription) {
      console.log("SENDING INSTRUCTIONS EMAIL TO: ", meeting.contact.email);
      await notification.sendEmail({
          emailAlias: companyName,
          to: meeting.contact.email,
          subject: '[Instructions] Tour instructions',
          body: emailBody
      });

      await Meeting.updateOne({ _id: meeting._id }, {
          $set: { 'instructions.sent': true, 'instructions.ts': new Date().getTime() }
      });
      console.log('Instructions sent for meeting:', meeting._id);
  }
}

module.exports = {
  sendReminder,
  sendInstructions,
}
