/**
 * Kit Routes Module
 * This module sets up routes for handling operations related to kits, including querying kits, updating shipping addresses, and managing kit states.
 * It ensures that only authenticated users can access these routes, and some routes require admin role.
 * 
 * @module routes/kit
 */
const router = require('express').Router()
const ObjectId = require('mongoose').Types.ObjectId
const Hardware = require('../models/hardware')
const Order = require('../models/order')
const Kit = require('../models/kit')
const User = require('../models/user')
const Subscription = require('../models/subscription')
const PaymentMethod = require('../models/paymentMethod')
const { Shipment } = require('../models/shipment')
const { authMiddleware } = require('../middlewares/authMiddleware')
const adminRole = require('../middlewares/adminRole')
const stripe = require('stripe')(process.env.stripe_sk_live)
const { splitTags } = require('../helpers/utils')
const {
  associateHardware,
  registerKitToUser,
  closeKitToUser,
} = require('../helpers/kit')
const paymentMethod = require('../models/paymentMethod')

/**
 * Get kits associated with a specific order.
 * 
 * @name GetOrderKits
 * @function
 * @memberof module:routes/kit
 * @param {string} orderId - The ID of the order to fetch kits for.
 * @param {Object} req - Express request object, with user information from authMiddleware.
 * @param {Object} res - Express response object.
 * @returns {Array} An array of kits associated with the order.
 */
router.get('/order/:orderId?', authMiddleware, async (req, res) => {
  const { orderId } = req.params
  try {
    const kits = await Kit.find({ order: orderId }).populate('shipment', 'trackingUrl').populate('invoiceInfo')
    return res.status(200).json(kits)
  } catch (error) {
    console.log(error)
    throw new Error(`There has been an error searching the kits by order id: `, error)
  }
})

/**
 * Get all kits or kits by a specific state.
 * 
 * @name GetAllKits
 * @function
 * @memberof module:routes/kit
 * @param {Object} req - Express request object, with user information from authMiddleware.
 * @param {Object} res - Express response object.
 * @returns {Array|Object} An array of kits or a single kit object.
 */
router.get('/', authMiddleware, (req, res) => {
  let query = {}
  if (req.query.state) query.state = req.query.state

  if (req.user.isSuperAdmin == 'superadmin') delete query.user

  let Query = Kit.find(query)

  if (req.params.id && ObjectId.isValid(req.params.id)) {
    query._id = req.params.id
    Query = Kit.findOne(query)
  }

  Query.populate('hardware')

  Query.exec((err, hardware) => {
    if (err) {
      res.status(500).send(err);
    } else {
      res.status(200).json(hardware);
    }
  })
})

/**
 * Confirm a kit.
 * 
 * @name ConfirmKit
 * @function
 * @memberof module:routes/kit
 * @param {string} id - The ID of the kit to confirm.
 * @param {Object} req - Express request object, with user information from authMiddleware.
 * @param {Object} res - Express response object.
 * @returns {Object} The confirmed kit object.
 */
router.patch('/:id/confirm', authMiddleware, async (req, res) => {
  const { id: kitId } = req.params
  const { _id: userId } = req.user
  const {paymentMethodId} = req.body

  try {
    const kit = await Kit.findById(kitId).populate('shipment invoiceInfo').populate('user').select('stripeCustomerId').populate('order').select('paymentMethod')

    if (!kit) {
      res.status(401).send({ message: 'Kit not found' })
      return
    }

    if (!kit.invoiceInfo) {
      res.status(401).send({ message: "This kit doesn't have invoice information" })
      return
    }

    const { shipment, invoiceInfo, user, order } = kit

    const { stripeCustomerId: customer } = user
    const { paymentMethod } = order

    let subscription = await Subscription.findOne({
      invoiceInfo: invoiceInfo._id
    })

    if (!subscription) {
      console.log('creating new subcription')

      if (paymentMethod === 'credit card') {

        const userPaymentMethod = paymentMethodId ? await PaymentMethod.findById(paymentMethodId) : await PaymentMethod.findOne({
          user: userId
        });

        if (!userPaymentMethod) {
          console.log("NO PAYMENT METHOD")
          res.status(401).send({message: "This user doesn't have a payment method we can use"})
        }

        const stripeSubscription = await stripe.subscriptions.create({
          customer: customer,
          items: [{ price: 'price_1P9tcgFIjHkIJIZyUHKxYvse', quantity: 1 }],
          expand: ['latest_invoice.payment_intent'],
          proration_behavior: 'create_prorations',
          default_payment_method: userPaymentMethod.stripe.paymentMethod,
          metadata: {
            userId,
            orderId: kit.order,
            paymentMethod: 'credit card',
            invoiceInfoId: invoiceInfo._id,
          },
        });

        console.log("Subscription created: ", stripeSubscription);

      } else {

        const stripeSubscription = await stripe.subscriptions.create({
          customer: customer.id,
          items: [{ price: 'price_1P9tcgFIjHkIJIZyUHKxYvse', quantity: 1 }],
          payment_behavior: 'default_incomplete',
          expand: ['latest_invoice.payment_intent'],
          proration_behavior: 'create_prorations',
          metadata: {
            userId,
            orderId: kit.order,
            paymentMethod: 'check',
            invoiceInfoId: invoiceInfo._id,
          },
        });

        console.log("Subscription created: ", stripeSubscription);

        let newInvoice = await stripe.invoices.create({
          customer: customer.id,
          subscription: stripeSubscription.id,
          auto_advance: false,
          customer_name: invoiceInfo.name,
          metadata: {
            userId,
          }
        });

        if (newInvoice.status === 'draft') {
          await stripe.invoices.finalizeInvoice(newInvoice.id, {
            auto_advance: true,
          });
        }

      }

    } else {
      const { subscription: stripeSubscription, customer } = subscription.stripe
      const { paymentMethod } = subscription
      const subscriptionObject = await stripe.subscriptions.retrieve(stripeSubscription);
      const subscriptionItemId = subscriptionObject.items.data[0].id;

      if (subscription.status === 'active') {
        console.log("This user has an active subscription")

        if (paymentMethod === 'check') {
          await stripe.subscriptions.update(stripeSubscription, {
            items: [{
              id: subscriptionItemId,
              quantity: subscriptionObject.items.data[0].quantity + 1,
            }],
            proration_behavior: 'create_prorations',
            proration_date: Math.floor((shipment ? new Date(shipment.deliveredAt).getTime() + 50000 : Date.now() + 5000) / 1000),
          });

          const newInvoice = await stripe.invoices.create({
            customer,
            subscription: stripeSubscription,
            auto_advance: subscription.paymentMethod === 'credit card',
            metadata: {
              userId,
            }

          });

          if (subscription.paymentMethod === 'check' && newInvoice.status === 'draft') {
            await stripe.invoices.finalizeInvoice(newInvoice.id, {
              auto_advance: true,
            });
          }
        }

      } else if (subscription.status === 'pending') {
        console.log("This user has an pending subscription")

        await stripe.subscriptions.update(stripeSubscription, {
          billing_cycle_anchor: 'now',
        });

        subscription.state = 'active';
        subscription
        await subscription.save()

      }

    }

    kit.state = 'confirmed'
    kit.save()

    res.status(200).json(kit)
  } catch (error) {
    console.log(error)
  }
})

      if (err) {
        res.status(500).send(err);

})
