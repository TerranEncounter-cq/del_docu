/**
 * Handles requests related to keypads.
 * @module Controller/keypad
 */
const ObjectId = require('mongoose').Types.ObjectId;
const { v4: uuidv4 } = require('uuid');
const Hardware = require('../models/hardware');
const switchbotController = require('../helpers/switchbot');
const keypadController = require('../helpers/keypad');

/**
 * Creates a new passcode for a keypad.
 * @param {Object} req - The request object.
 * @param {Object} res - The response object.
 * @param {Function} next - The next middleware function.
 */
const createPasscode = (req, res, next) => {
    if (!ObjectId.isValid(req.params.id)) {
        return res.status(400).end();
    }

    let query = {
        _id: req.params.id,
        category: 'keypad',
        status: 'active'
    }
    Hardware
        .findOne(query)
        .populate('account', 'token secret')
        .exec((errHardware, hardware) => {
            if (errHardware) {
                console.log(errHardware);
            } else {
                const authData = {
                    token: hardware.account.token,
                    secret: hardware.account.secret,
                    t: Date.now(),
                    nonce: uuidv4()
                }
                switchbotController
                    .sendCommand(
                        hardware.keypad.id,
                        'createKey',
                        {
                            name: req.body.name,
                            type: 'permanent',
                            password: req.body.passcode
                        },
                        authData
                    )
                    .then(response => {
                        // find passcodes and add it to database
                        keypadController.passcodeList(hardware.keypad.id, authData)
                        .then((newPasscodes) => {
                            Hardware.findOneAndUpdate(
                                { _id: hardware._id },
                                {
                                    $set: {
                                        'keypad.passcodes': newPasscodes
                                    }
                                },
                                { new: true },
                                (errHardware) => {
                                    if (errHardware) {
                                        return res.status(500).send(err);
                                    } else {
                                        return res.status(200).send(newPasscodes);
                                    }
                                });
                        })
                        .catch((err) => {
                            return res.status(500).send(err);
                        })
                    }).catch((err) => {
                        return res.status(500).send(err);
                    })
            }
        })
};

/**
 * Deletes an existing passcode from a keypad.
 * @param {Object} req - The request object.
 * @param {Object} res - The response object.
 * @param {Function} next - The next middleware function.
 */
const deletePasscode = (req, res, next) => {
    if (!ObjectId.isValid(req.params.id)) {
        return res.status(400).end();
    }

    let query = {
        _id: req.params.id,
        category: 'keypad',
        status: 'active'
    }
    Hardware
        .findOne(query)
        .populate('account', 'token secret')
        .exec((errHardware, hardware) => {
            if (errHardware) {
                console.log(errHardware);
            } else {
                const authData = {
                    token: hardware.account.token,
                    secret: hardware.account.secret,
                    t: Date.now(),
                    nonce: uuidv4()
                }
                switchbotController
                    .sendCommand(
                        hardware.keypad.id,
                        'deleteKey',
                        {
                            id: req.body.id,
                        },
                        authData
                    )
                    .then(response => {
                            // find passcodes and remove deleted passcode
                            keypadController.passcodeList(hardware.keypad.id, authData)
                                .then((newPasscodes) => {
                                    newPasscodes = newPasscodes.filter(pc => pc.id != req.body.id);
                                    Hardware.findOneAndUpdate(
                                        { _id: hardware._id },
                                        {
                                            $set: {
                                                'keypad.passcodes': newPasscodes
                                            }
                                        },
                                        { new: true },
                                        (errHardware) => {
                                            if (errHardware) {
                                                return res.status(500).send(err);
                                            } else {
                                                return res.status(200).send(newPasscodes);
                                            }
                                        });
                                })
                                .catch((err) => {
                                    return res.status(500).send(err);
                                })
                    }).catch((err) => {
                        return res.status(500).send(err);
                    })
            }
        })
};
